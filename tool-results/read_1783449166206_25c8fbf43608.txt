    71→    71→async function callClaude(messages, systemPrompt, wantJson = false) {
    72→    72→  const res = await fetch("https://api.anthropic.com/v1/messages", {
    73→    73→    method: "POST",
    74→    74→    headers: { "Content-Type": "application/json" },
    75→    75→    body: JSON.stringify({ model: "claude-sonnet-4-6", max_tokens: 1000, system: systemPrompt, messages }),
    76→    76→  });
    77→    77→  const data = await res.json();
    78→    78→  const text = (data.content || []).map((b) => b.text || "").join("\n");
    79→    79→  if (wantJson) {
    80→    80→    const cleaned = text.replace(/```json|```/g, "").trim();
    81→    81→    try { return JSON.parse(cleaned); } catch { return null; }
    82→    82→  }
    83→    83→  return text;
    84→    84→}
